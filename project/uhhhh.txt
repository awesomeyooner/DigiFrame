#include <QApplication>
#include <QWidget>
#include <QPainter>
#include <QScreen>
#include <QIcon>

class ImageWidget : public QWidget {
public:
    ImageWidget(const QString& imagePath, QWidget* parent = nullptr) 
        : QWidget(parent) {
        // Load image
        if (!m_image.load(imagePath)) {
            qFatal("Failed to load image: %s", qUtf8Printable(imagePath));
        }
        
        // Configure window
        setWindowState(Qt::WindowFullScreen);
        setWindowFlags(Qt::FramelessWindowHint);
        setAttribute(Qt::WA_OpaquePaintEvent);
    }

protected:
    void paintEvent(QPaintEvent*) override {
        QPainter painter(this);
        painter.setRenderHint(QPainter::SmoothPixmapTransform);
        
        // Calculate scaled dimensions with aspect ratio
        QSize scaledSize = m_image.size().scaled(
            size(), 
            Qt::KeepAspectRatio
        );
        
        // Calculate centering position
        int x = (width() - scaledSize.width()) / 2;
        int y = (height() - scaledSize.height()) / 2;
        QRect targetRect(x, y, scaledSize.width(), scaledSize.height());
        
        // Draw image (with letterboxing if needed)
        painter.fillRect(rect(), Qt::black);  // Background
        painter.drawImage(targetRect, m_image);
        
        // === DRAW OVERLAYS ===
        // 1. Draw text
        painter.setPen(Qt::white);
        painter.setFont(QFont("Arial", 24, QFont::Bold));
        painter.drawText(rect().adjusted(0, 0, 0, -50), 
                        Qt::AlignBottom | Qt::AlignHCenter,
                        "Sample Overlay Text");
        
        // 2. Draw icon
        QIcon icon("path/to/icon.png");
        int iconSize = 64;
        icon.paint(&painter, 
                  width() - iconSize - 20, 
                  height() - iconSize - 20,
                  iconSize, iconSize);
        
        // 3. Draw timestamp
        painter.setPen(Qt::yellow);
        painter.drawText(20, 40, QDateTime::currentDateTime().toString());
        
        // 4. Draw semi-transparent rectangle
        painter.setBrush(QColor(0, 0, 0, 150));  // Semi-black
        painter.setPen(Qt::NoPen);
        painter.drawRect(0, 0, 300, 80);
    }

    void keyPressEvent(QKeyEvent* event) override {
        if (event->key() == Qt::Key_Escape) qApp->quit();
    }

private:
    QImage m_image;
};

int main(int argc, char *argv[]) {
    QApplication app(argc, argv);
    ImageWidget widget("path/to/image.jpg");
    widget.show();
    return app.exec();
}
